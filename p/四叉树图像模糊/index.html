#pragma once
#include <vector>
#include <cmath>

using namespace std;

class QuadTree {
private:
    int x, y, w, h;                     //å››å‰æ ‘çš„ä½ç½®å’Œå¤§å°
    double ColorAverage, EColor;        //å‡å€¼å’Œæ–¹å·®
    bool Divided;                       //æ˜¯å¦åˆ†å‰²
    QuadTree* topLeft;
    QuadTree* topRight;
    QuadTree* bottomLeft;
    QuadTree* bottomRight;

public:
    QuadTree(int x, int y, int w, int h, int*** ImageData, int Width, int Height) :
        x(x), y(y), w(w), h(h), ColorAverage(0), EColor(0), Divided(false), topLeft(nullptr), topRight(nullptr), bottomLeft(nullptr), bottomRight(nullptr)
    {
        int l = w * h;
        vector<double> colord(l);
        double sum = 0.0;

        for (int i = y; i < (y + h); i++) {
            for (int j = x; j < (x + w); j++) {
                colord[(i - y) * w + (j - x)] = sqrt((pow(ImageData[i][j][0], 2) + pow(ImageData[i][j][1], 2) + pow(ImageData[i][j][2], 2)) / 3);
                sum += colord[(i - y) * w + (j - x)];
            }
        }

        ColorAverage = sum / l;

        for (int i = 0; i < l; i++) EColor += pow(colord[i] - ColorAverage, 2);
        EColor = sqrt(EColor / l);
    }

    ~QuadTree() {
        delete topLeft;
        delete topRight;
        delete bottomLeft;
        delete bottomRight;
    }

    void Divide(int*** ImageData, int Width, int Height, double key) {
        if ((w == 1 && h == 1) || (EColor <= key)) return;

        Divided = true;
        int newW = w / 2;
        int newH = h / 2;

        topLeft = new QuadTree(x, y, newW, newH, ImageData, Width, Height);
        topRight = new QuadTree(x + newW, y, newW, newH, ImageData, Width, Height);
        bottomLeft = new QuadTree(x, y + newH, newW, newH, ImageData, Width, Height);
        bottomRight = new QuadTree(x + newW, y + newH, newW, newH, ImageData, Width, Height);

        topLeft->Divide(ImageData, Width, Height, key);
        topRight->Divide(ImageData, Width, Height, key);
        bottomLeft->Divide(ImageData, Width, Height, key);
        bottomRight->Divide(ImageData, Width, Height, key);
    }

    void MeanBlur(int*** ImageData, int Width, int Height) {
        if (Divided) {
            topLeft->MeanBlur(ImageData, Width, Height);
            topRight->MeanBlur(ImageData, Width, Height);
            bottomLeft->MeanBlur(ImageData, Width, Height);
            bottomRight->MeanBlur(ImageData, Width, Height);
        }
        else {
            int l = w * h;
            int* RGB = new int[3] {0, 0, 0};

            for (int i = y; i < y + h; i++) {
                for (int j = x; j < x + w; j++) {
                    for (int k = 0; k < 3; k++) {
                        RGB[k] += ImageData[i][j][k];
                    }
                }
            }
            for (int k = 0; k < 3; k++) RGB[k] /= l;

            for (int i = y; i < (y + h); i++) {
                for (int j = x; j < (x + w); j++) {
                    for (int k = 0; k < 3; k++) {
                        ImageData[i][j][k] = RGB[k];
                    }
                }
            }
        }
    }
};

#include <vector>
#define PI       3.14159265358979323846

void GaussianBlur(int*** ImageData, int Width, int Height, int blurLevel) {
    if (blurLevel < 1) blurLevel = 1;                                                       // ç¡®ä¿æ¨¡ç³Šç¨‹åº¦è‡³å°‘ä¸º 1
    const int kernelSize = 2 * blurLevel + 1;                                               // å†…æ ¸å¤§å°ä¸º (2 * blurLevel + 1)ï¼š1ä¸ºä¸­å¿ƒï¼Œblurlevelä¸ºåŠå¾„
    const double sigma = static_cast<double>(blurLevel);                                    // æ ¹æ®æ¨¡ç³Šç¨‹åº¦è®¾ç½® sigma
    std::vector<std::vector<double>> kernel(kernelSize, std::vector<double>(kernelSize));

    //åˆ›å»ºé«˜æ–¯å†…æ ¸
    double sum = 0.0;
    int halfSize = kernelSize / 2;
    for (int i = -halfSize; i <= halfSize; i++) {
        for (int j = -halfSize; j <= halfSize; j++) {
            kernel[i + halfSize][j + halfSize] = exp(-(i * i + j * j) / (2.0 * sigma * sigma)) / (2.0 * PI * sigma * sigma);
            sum += kernel[i + halfSize][j + halfSize];
        }
    }
    //å½’ä¸€
    for (int i = 0; i < kernelSize; i++) {
        for (int j = 0; j < kernelSize; j++) {
            kernel[i][j] /= sum;
        }
    }

    // åˆ›å»ºæ–°çš„å›¾åƒæ•°æ®
    int*** newImageData = new int** [Height];
    for (int i = 0; i < Height; i++) {
        newImageData[i] = new int* [Width];
        for (int j = 0; j < Width; j++) {
            newImageData[i][j] = new int[3];
        }
    }

    // è¿›è¡Œæ¨¡ç³Š
    for (int i = 0; i < Height; i++)
        for (int j = 0; j < Width; j++)
            for (int k = 0; k < 3; k++){
                double blurredPixel = 0.0;
                for (int x = -halfSize; x <= halfSize; x++) {
                    for (int y = -halfSize; y <= halfSize; y++) {
                        int neighborX = j + x;
                        int neighborY = i + y;
                        if (neighborX >= 0 && neighborX < Width && neighborY >= 0 && neighborY < Height) {
                            blurredPixel += ImageData[neighborY][neighborX][k] * kernel[x + halfSize][y + halfSize];
                        }
                    }
                }
                newImageData[i][j][k] = (int)(blurredPixel);
            }

    for (int i = 0; i < Height; i++) {
        for (int j = 0; j < Width; j++) {
            for (int k = 0; k < 3; k++) {
                ImageData[i][j][k] = newImageData[i][j][k];
            }
        }
    }

    for (int i = 0; i < Height; i++) {
        for (int j = 0; j < Width; j++)
            delete[] newImageData[i][j];
        delete[] newImageData[i];
    }
    delete[] newImageData;

}
